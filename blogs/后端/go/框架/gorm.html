    type Product struct {
  gorm.Model
  Code  string
  Price uint
}

func main() {
  db, err := gorm.Open(sqlite.Open("test.db"), &gorm.Config{})
  if err != nil {
    panic("failed to connect database")
  }

  // 迁移 schema
  db.AutoMigrate(&Product{})

  // Create
  db.Create(&Product{Code: "D42", Price: 100})

  // Read
  var product Product
  db.First(&product, 1) // 根据整型主键查找
  db.First(&product, "code = ?", "D42") // 查找 code 字段值为 D42 的记录

  // Update - 将 product 的 price 更新为 200
  db.Model(&product).Update("Price", 200)
  // Update - 更新多个字段
  db.Model(&product).Updates(Product{Price: 200, Code: "F42"}) // 仅更新非零值字段
  db.Model(&product).Updates(map[string]interface{}{"Price": 200, "Code": "F42"})

  // Delete - 删除 product
  db.Delete(&product, 1)
}

    # 链式操作

    GORM 将方法分为三大类： Chain Methods, Finisher Methods, and New Session Methods。

    # Chain Methods

    用于修改或追加目前 Clauses 的 Statement。 一些常见的链式方法包括：

    • Where
    • Select
    • Omit
    • Joins
    • Scopes
    • Preload
    • Raw (Raw 不能与其他可链接的方法一起构建 SQL)

    # Finisher Methods

    结束方法是即时的，执行生成和运行 SQL 命令的注册回调。 常见的结束方法:

    • Create
    • First
    • Find
    • Take
    • Save
    • Update
    • Delete
    • Scan
    • Row
    • Rows

    # 链式安全

    链式第一次查出来的数据可能会影响第二次查询，所以每次查询最好是一次新的会话，否则上一次的查询会污染下一次的查询。

    // s7更好理解了，长度和容量都是9

// MulUintptr第一行代码的判断条件可以简化成  a|b < 1 << 4 * 8 || a == 0，也即 a|b < 2^32|| a== 0 
// 其中a==0的情况，a*b 一定不溢出
// a|b < 2^32的情况下，代表a或者b都小于2^32，也就是说a * b 一定小于 2^64-1，也不会溢出
// 现在翻译一下MulUintptr的逻辑：
// 1. 如果a、b乘积不溢出或者a == 0，直接返回乘积a*b和false
// 2. 其它情况，overflow = a * b > MaxUintptr(uintptr最大值)，返回a * b和true
func MulUintptr(a, b uintptr) (uintptr, bool) {
    // goarch.PtrSize的值与平台相关，在64位系统下这个值是8
    if a|b < 1<<(4*goarch.PtrSize) || a == 0 {
        return a * b, false
    }
    overflow := b > MaxUintptr/a
    return a * b, overflow
}

// uintptr(0)：这将无符号整数 0 转换为 uintptr 类型。uintptr 是一个无符号整数类型，其大小等于指针的大小，通常是 32 位或 64 位。
// ^uintptr(0)：这是对无符号整数 0 进行按位取反操作，得到所有位都为 1 的值。在 64 位系统上，结果为 0xffffffffffffffff，在 32 位系统上，结果为 0xffffffff。
// ^uintptr(0) >> 63：这是将上一步得到的结果右移 63 位，结果是 64 位系统下为 1（0x1），32 位系统下为 0（0x0）。
// 4 << (^uintptr(0) >> 63)：这是将上一步得到的结果作为位移量，对数字 4 进行左移运算。如果在 64 位系统上，结果是 4 << 1，即 8；如果在 32 位系统上，结果是 4 << 0，即 4。
const MaxUintptr = ^uintptr(0)
const PtrSize = 4 << (^uintptr(0) >> 63)

    # 扩容切片

    # 扩容规则

    // nextslicecap 计算下一个合适的片长度
func nextslicecap(newLen, oldCap int) int {
    newcap := oldCap
    doublecap := newcap + newcap
    // 如果新长度大于两倍的旧容量，直接扩容到新长度
    if newLen > doublecap {
        return newLen
    }
    const threshold = 256
    // 如果旧容量小于256，则直接扩容到两倍
    if oldCap < threshold {
        return doublecap
    }
    for {
        // 对于旧容量大于等于256，每次扩容趋近于1.25倍直到满足新容量
        // 底层还有内存对齐
        newcap += (newcap + 3*threshold) >> 2
        // 如果新容量已经达到或超过所需长度，退出循环
        if uint(newcap) >= uint(newLen) {
            break
        }
    }
    // 防止溢出，如果溢出，则返回新长度
    if newcap <= 0 {
        return newLen
    }
    return newcap
}

    # 扩容机制总结

    一、go1.18 之前：

    1. 如果期望容量大于当前容量的两倍就会使用期望容量
    2. 如果当前切片的长度小于 1024 就会将容量翻倍
    3. 如果当前切片的长度大于 1024 就会每次增加 25% 的容量，直到新容量大于期望容量

    二、go1.18 之后：

    1. 如果期望容量大于当前容量的两倍就会使用期望容量
    2. 如果当前切片的长度小于阈值（默认 256）就会将容量翻倍；
    3. 如果当前切片的长度大于等于阈值（默认 256），就会每次增加 25% 的容量，基准是 newcap + 3*threshold，直到新容量大于期望容量；

    如果只选择翻倍的扩容策略，那么对于较大的切片来说，现有的方法可以更好的节省内存 如果只选择每次系数为1.25的扩容策略，那么对于较小的切片来说扩容会很低效 之所以选择一个小于2的系数，在扩容时被释放的内存块会在下一次扩容时更容易被重新利用

    # 内存对齐

    func growslice(oldPtr unsafe.Pointer, newLen, oldCap, num int, et *_type) slice {
    oldLen := newLen - num  // 计算旧长度

    // 如果启用了竞态检测，则进行读操作的竞态检测
    if raceenabled {
        callerpc := getcallerpc()
        racereadrangepc(oldPtr, uintptr(oldLen*int(et.Size_)), callerpc, abi.FuncPCABIInternal(growslice))
    }
    // 如果启用了内存清理检测（MSan），则进行读操作的检测
    if msanenabled {
        msanread(oldPtr, uintptr(oldLen*int(et.Size_)))
    }
    // 如果启用了地址清理检测（ASan），则进行读操作的检测
    if asanenabled {
        asanread(oldPtr, uintptr(oldLen*int(et.Size_)))
    }

    // 如果新长度小于0，触发panic
    if newLen < 0 {
        panic(errorString("growslice: len out of range"))
    }

    // 如果元素类型大小为0
    if et.Size_ == 0 {
        // append不应该创建指针为nil但长度不为0的slice
        // 在这种情况下，我们假设append不需要保留旧的指针
        return slice{unsafe.Pointer(&zerobase), newLen, newLen}
    }

    newcap := oldCap  // 初始化新的容量为旧的容量
    doublecap := newcap + newcap  // 计算容量的两倍
    // 如果新长度超过了两倍容量，则将新容量设置为新长度
    if newLen > doublecap {
        newcap = newLen
    } else {
        const threshold = 256  // 定义一个容量阈值
        // 如果旧容量小于阈值，则将新容量设置为两倍容量
        if oldCap < threshold {
            newcap = doublecap
        } else {
            // 检查新容量是否为正数以检测溢出并防止无限循环
            for 0 < newcap && newcap < newLen {
                // 从对小slice的2倍增长过渡到对大slice的1.25倍增长
                // 这个公式提供了一个相对平滑的过渡
                newcap += (newcap + 3*threshold) / 4
            }
            // 如果新容量计算溢出，则将新容量设置为请求的容量
            if newcap <= 0 {
                newcap = newLen
            }
        }
    }

    var overflow bool
    var lenmem, newlenmem, capmem uintptr

    // 针对常见的et.Size_值进行优化
    // 对于大小为1的类型，不需要进行任何除法/乘法
    // 对于指针大小的类型，编译器将优化除法/乘法为常量移位
    // 对于2的幂大小的类型，使用变量移位操作
    switch {
    case et.Size_ == 1:
        lenmem = uintptr(oldLen)  // 计算旧数据的内存大小
        newlenmem = uintptr(newLen)  // 计算新数据的内存大小
        capmem = roundupsize(uintptr(newcap))  // 计算新容量的内存大小
        overflow = uintptr(newcap) > maxAlloc  // 检查容量是否溢出
        newcap = int(capmem)
    case et.Size_ == goarch.PtrSize:
        lenmem = uintptr(oldLen) * goarch.PtrSize
        newlenmem = uintptr(newLen) * goarch.PtrSize
        capmem = roundupsize(uintptr(newcap) * goarch.PtrSize)
        overflow = uintptr(newcap) > maxAlloc/goarch.PtrSize
        newcap = int(capmem / goarch.PtrSize)
    case isPowerOfTwo(et.Size_):
        var shift uintptr
        // 针对64位架构，使用位运算生成更优代码
        if goarch.PtrSize == 8 {
            shift = uintptr(sys.TrailingZeros64(uint64(et.Size_))) & 63
        } else {
            shift = uintptr(sys.TrailingZeros32(uint32(et.Size_))) & 31
        }
        lenmem = uintptr(oldLen) << shift  // 左移操作来计算内存大小
        newlenmem = uintptr(newLen) << shift
        capmem = roundupsize(uintptr(newcap) << shift)
        overflow = uintptr(newcap) > (maxAlloc >> shift)
        newcap = int(capmem >> shift)
        capmem = uintptr(newcap) << shift
    default:
        lenmem = uintptr(oldLen) * et.Size_  // 默认情况下按元素大小计算内存
        newlenmem = uintptr(newLen) * et.Size_
        capmem, overflow = math.MulUintptr(et.Size_, uintptr(newcap))
        capmem = roundupsize(capmem)
        newcap = int(capmem / et.Size_)
        capmem = uintptr(newcap) * et.Size_
    }

    // 除了检查溢出，还需要检查capmem > maxAlloc，以防止溢出导致的段错误
    if overflow || capmem > maxAlloc {
        panic(errorString("growslice: len out of range"))
    }

    var p unsafe.Pointer
    // 如果元素类型不包含指针
    if et.PtrBytes == 0 {
        p = mallocgc(capmem, nil, false)  // 分配内存，容量为capmem
        // append() 调用growslice时，会覆盖从oldLen到newLen的区域
        // 因此只需清理不会被覆盖的部分
        memclrNoHeapPointers(add(p, newlenmem), capmem-newlenmem)
    } else {
        // 不能使用rawmem，因为GC可能扫描未初始化的内存
        p = mallocgc(capmem, et, true)
        // 如果有旧数据并且写屏障已启用
        if lenmem > 0 && writeBarrier.enabled {
            // 只处理oldPtr中的指针，因为我们知道目标slice p
            // 只包含nil指针，因为它在分配期间已经被清空
            bulkBarrierPreWriteSrcOnly(uintptr(p), uintptr(oldPtr), lenmem-et.Size_+et.PtrBytes)
        }
    }
    // 将旧数据移动到新分配的内存中
    memmove(p, oldPtr, lenmem)

    // 返回新的slice，包含新的指针、新的长度和新的容量
    return slice{p, newLen, newcap}
}

    # 内存对齐分析

    内存对齐提高CPU访问内存的效率

    1. CPU访问内存的方式：
      • CPU通常按字（word）访问内存，现代的64位CPU一次可以读取或写入8个字节（64位）。如果数据存储在内存中是对齐的（aligned），那么CPU可以在一次内存访问中读取或写入整个数据。
    2. 内存对齐的意义：
      • 如果数据未对齐（unaligned），比如一个64位的数据分布在两个不同的8字节边界上，CPU需要进行两次内存访问才能完整读取或写入这个数据。这样不仅增加了内存访问的时间，还可能引发性能问题。
    3. 内存对齐的规则：
      • 内存对齐通常遵循某些规则，比如32位数据类型应该在4字节边界上对齐，64位数据类型应该在8字节边界上对齐。编译器会自动在结构体或类中插入填充字节（padding），使得每个数据成员都按其对齐要求存储。
    // Returns size of the memory block that mallocgc will allocate if you ask for the size.
func roundupsize(size uintptr) uintptr {
    if size < _MaxSmallSize {
        if size <= smallSizeMax-8 {
            return uintptr(class_to_size[size_to_class8[divRoundUp(size, smallSizeDiv)]])
        } else {
            return uintptr(class_to_size[size_to_class128[divRoundUp(size-smallSizeMax, largeSizeDiv)]])
        }
    }
    if size+_PageSize < size {
        return size
    }
    return alignUp(size, _PageSize)
}

    # 拷贝切片

    # 避坑总结

    # 引用类型

    切片是引用类型，与数组是共享存储的，切片改变，数组也改变。

    # 参数传递

    Go的slice类型中包含了一个array指针以及len和cap两个int类型的成员。

    Go中的参数传递实际都是值传递，将slice作为参数传递时，函数中会创建一个slice参数的副本，这个副本同样也包含array,len,cap这三个成员。

    副本中的array指针与原slice指向同一个地址，所以当修改副本slice的元素时，原slice的元素值也会被修改。但是如果修改的是副本slice的len和cap时，原slice的len和cap仍保持不变。

    如果在操作副本时由于扩容操作导致重新分配了副本slice的array内存地址，那么之后对副本slice的操作则完全无法影响到原slice，包括slice中的元素。

    # 遍历切片

    对于slice，时刻想着对slice做了修改后，slice里的3个字段：指针，长度，容量是怎么变的。

    对切片x做range遍历，实际上是对x的拷贝(假设为y)做range遍历，y的值(包括y结构体里指向底层数组的指针的值，y的长度和容量)都在执行for循环前确定下来了。

    # 3. 使用copy函数

    package main

import "fmt"

// 使用copy函数删除指定索引的元素
func removeAtIndex(slice []int, index int) []int {
    // 如果索引超出范围，直接返回原切片
    if index < 0 || index >= len(slice) {
        return slice
    }
    
    // 使用copy函数将后面的元素向前移动
    copy(slice[index:], slice[index+1:])
    
    // 返回新的切片，长度减1
    return slice[:len(slice)-1]
}

func main() {
    arr := []int{1, 2, 3, 4, 5}
    index := 2 // 要删除的元素索引

    // 删除元素
    arr = removeAtIndex(arr, index)

    fmt.Println(arr) // 输出: [1 2 4 5]
}
